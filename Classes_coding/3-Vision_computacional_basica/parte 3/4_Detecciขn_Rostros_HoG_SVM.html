# Extracci√≥n de regiones ("patches") de las im√°genes negativas
from sklearn.feature_extraction.image import PatchExtractor
from skimage import transform

# Funci√≥n para extraer los "patches"
def extract_patches(I, N, scale=1.0, psize=None):
    # Definir el tama√±o del patch
    patch_size = tuple((scale*np.array(psize)).astype(int))
    # Extractor de patches
    pextractor = PatchExtractor(patch_size=patch_size, max_patches=N, random_state=0)
    # Extracci√≥n de patches a la imagen 
    patches = pextractor.transform(I[np.newaxis])
    # Si patches de tama√±o diferente que el tama√±o de los patches positivos:
    # re-escalar al tama√±o de los patches positivos
    if scale != 1:
        patches = np.array([transform.resize(patch, psize) for patch in patches])
    return patches

# Im√°genes negativas: se usa patches de 0.5, 1.0 y 2.0 el tama√±o del patch positivo
I_negative = np.vstack([extract_patches(im, 1000, scale, psize=I_positive[0].shape) 
                        for im in Imgs for scale in [0.5, 1.0, 2.0]] )

# Tama√±o del conjunto "negativo"
print("Tama√±o de im√°genes negativas:", I_negative.shape)

# Define "ventanas deslizantes" en la imagen
def sliding_window(I, scale=1.0, patch_size=None, istep=2, jstep=2):
    Ni, Nj = (int(scale*s) for s in patch_size)
    for i in range(0, I.shape[0]-Ni, istep):
        for j in range(0, I.shape[1]-Ni, jstep):
            patch = I[i: i+Ni, j:j+Nj]
            if scale != 1:
                patch = transform.resize(patch, patch_size)
            yield (i,j), patch

# Obtenci√≥n de patches usando "ventanas deslizantes" a lo largo de la imagen
indices, patches = zip(*sliding_window(I_test, patch_size=I_positive[0].shape))

# Aplicaci√≥n de HOG a cada patch: X de prueba
X_test = np.array([feature.hog(patch) for patch in patches])

print("N√∫mero de patches de prueba(Xtest):", len(patches))
print("Tama√±o de cada vector de atributos HOG:", X_test.shape[1])
print("Tama√±o de X de prueba:", X_test.shape)
